mail_sorter/
│
├── data/
│   ├── inbox/               (100 исходных писем .txt)
│   ├── processed/           (создаётся автоматически, внутри папки категорий)
│   ├── config/
│   │   └── categories.json  (определяет категории и правила приоритета)
│   │   └── candidate_keywords.json  (генерируемый файл с поиском возможны ключевых слов)
│   └── reports/             (результаты работы)
│       ├── summary.txt
│       └── log.txt
│
├── src/
│   ├── processing/              (приводит исодные данные в стандартизированный вид)
│       └── parser.py
│       └── cleaner.py           # приведение текста к нижнему регистру, удаление мусора
│   └── analyzer/                (сортирует письма на основе обработанны данны)
│       └── keyword_extractor.py # выделяет все ключевые слова по их частоте
│       └── classifier.py        # загрузка правил из config, применение к письму
│       └── file_manager.py      # перемещение файлов, создание папок
│       └── reporter.py          # генерация отчётов
│   └── main.py                  # соединение всех функций
│
├── extensions/
│   └── vizualizator.py      # tkinter для красивой визуализации исходных и итоговых данных
│
├── tests/
│   ├── test_parser.py
│   ├── test_classifier.py
│   ├── test_file_manager.py
│   └── fixtures/            # тестовые письма (5-10 штук)
│
├── run.sh                   # bash-скрипт запуска
├── requirements.txt         # зависимости (без ML для базы, но с опцией)
├── README.md
└── .gitignore
